Step 2 — Grid renderer: strict visible row/column logic
renderGrid.js▼
This is the "deep-dive" model. Only stack_bond, running_bond, stretcher_bond, flemish_bond use this. Tile size is derived from the canvas divided by rows/cols. Seam wrapping is mandatory for offset bonds.
// renderGrid.js
// Used only by patterns with rowColMode === 'grid'

export function renderGrid(ctx, canvasW, canvasH, state, materialImg) {
  const { rows, cols, pattern } = state;
  const offsetFrac = pattern.offsetFraction ?? 0;

  const jointV = state.joints.vertical_px;    // already converted to px
  const jointH = state.joints.horizontal_px;

  // Tile size derived from canvas ÷ rows/cols
  // This is the STRICT VISIBLE model:
  // rows = how many tile courses fit in canvasH
  // cols = how many tile units fit in canvasW (before offset)
  const cellW = (canvasW - cols * jointV) / cols;
  const cellH = (canvasH - rows * jointH) / rows;

  ctx.clearRect(0, 0, canvasW, canvasH);

  for (let r = 0; r < rows; r++) {
    // Row offset for bond patterns
    const rowOffset = (r % 2 === 1)
      ? (cellW + jointV) * offsetFrac
      : 0;

    // Draw cols + 1 tiles per row to handle seam wrap
    for (let c = -1; c <= cols; c++) {
      const x = c * (cellW + jointV) + rowOffset;
      const y = r * (cellH + jointH);

      // Skip tiles fully outside canvas
      if (x + cellW < 0 || x > canvasW) continue;

      // Apply tone variation (seeded RNG — not Math.random())
      const seed = seededRandom(r * 9973 + c * 9871);
      applyToneVariation(ctx, state.toneVariation, seed, () => {
        ctx.drawImage(materialImg, x, y, cellW, cellH);
      });
    }
  }

  // Draw joints as a separate pass AFTER all tiles
  drawGridJoints(ctx, canvasW, canvasH, rows, cols, cellW, cellH, jointV, jointH, offsetFrac, state);
}

// Seam wrap: for offset bonds, tiles overflow the right edge.
// Canvas clips them automatically on the right.
// But the matching left-edge fragment must be drawn explicitly.
// That is handled by c = -1 in the loop above:
//   when c = -1, x = -1*(cellW+jointV) + rowOffset
//   for rowOffset = cellW/2: x = -cellW/2  ← draws the right-edge fragment at left
The seam wrap is the single most important line: starting the column loop at c = -1 (not 0) automatically handles the left-side bleed for offset bonds. No special case code needed.
Step 3 — Module renderer: leave existing SVG logic untouched
renderModule.js▼
Module patterns (herringbone, basketweave, etc.) keep their existing render functions. The only change is that the main render dispatcher checks rowColMode before calling them. No changes to the SVG module internals.
// render.js — main dispatcher

import { PATTERNS } from './patternRegistry.js';
import { renderGrid } from './renderGrid.js';

export function render(canvas, state) {
  const ctx = canvas.getContext('2d');
  const pattern = PATTERNS[state.pattern.id];

  if (!pattern) return;

  if (pattern.rowColMode === 'grid') {
    // Direct grid renderer — rows = courses, cols = units
    renderGrid(ctx, canvas.width, canvas.height, state, state.material.img);
  } else {
    // Module renderer — rows/cols = repeat module count
    // Call the pattern's own render function, UNCHANGED
    pattern.render(ctx, canvas.width, canvas.height, state, state.material.img);
  }
}
Step 4 — Canvas size formula: how rows/cols set canvasW/H
canvasSize.js▼
This is what drives the real-world size readout ("32 × 32 inches"). For grid patterns, canvas size = rows/cols × tile real-world size. For module patterns, canvas size = rows/cols × module real-world size. Same formula, different meaning of the unit.
// canvasSize.js

export function computeCanvasRealWorldSize(state) {
  const { rows, cols, pattern: patternState } = state;
  const p = PATTERNS[patternState.id];

  if (p.rowColMode === 'grid') {
    // Canvas real-world size = rows × (tileH + jointH) etc.
    const totalW = cols * (state.tile.width  + state.joints.vertical);
    const totalH = rows * (state.tile.height + state.joints.horizontal);
    return { width: totalW, height: totalH };   // in mm or inches

  } else {
    // Canvas real-world size = rows/cols × module size
    // Module size is pattern-specific (e.g. herringbone module = 2 tiles wide × 1 tile tall)
    const modW = p.moduleWidth(state);    // function defined per pattern
    const modH = p.moduleHeight(state);
    return {
      width:  cols * modW,
      height: rows * modH,
    };
  }
}

Step 5 — UI label switching based on rowColMode
sidebar UI▼
When the user switches patterns, the sidebar labels and help text update to match the rowColMode. No functional change to the inputs — just the label and tooltip text.
// patternControls.js  — runs whenever state.pattern.id changes

function updateRowColLabels(patternId) {
  const p = PATTERNS[patternId];
  const rowLabel  = document.getElementById('label-rows');
  const colLabel  = document.getElementById('label-cols');
  const rowHelp   = document.getElementById('help-rows');
  const colHelp   = document.getElementById('help-cols');

  if (p.rowColMode === 'grid') {
    rowLabel.textContent = 'Rows';
    colLabel.textContent = 'Columns';
    rowHelp.textContent  = 'Number of visible tile courses in one repeat';
    colHelp.textContent  = 'Number of visible tile units per course';
  } else {
    rowLabel.textContent = 'Rows';
    colLabel.textContent = 'Columns';
    rowHelp.textContent  = 'Number of pattern modules vertically in one repeat';
    colHelp.textContent  = 'Number of pattern modules horizontally in one repeat';
  }
}

Step 6 — Flemish bond special case (grid but two tile types per row)
renderFlemish.js▼
Flemish bond is rowColMode = 'grid' but each row alternates stretcher (long) and header (short/square) tiles. Cols in this context = number of stretcher+header pairs per row. The grid renderer can't handle this with a single tile size — it needs its own function.
// renderFlemish.js

export function renderFlemish(ctx, canvasW, canvasH, state, materialImg) {
  const { rows, cols } = state;
  const jointV = state.joints.vertical_px;
  const jointH = state.joints.horizontal_px;

  // In Flemish bond, cols = number of stretcher+header PAIRS
  // Each pair = 1 stretcher (2× width) + 1 header (1× width) + 2 joints
  // So: canvasW = cols × (stretcherW + headerW + 2×jointV)
  // stretcherW = 2 × headerW  (standard brick ratio)
  const headerW  = (canvasW - cols * 2 * jointV) / (cols * 3);  // 1 unit
  const stretcherW = headerW * 2;                                // 2 units
  const cellH = (canvasH - rows * jointH) / rows;

  for (let r = 0; r < rows; r++) {
    const isOffset = r % 2 === 1;
    let x = isOffset ? -(headerW + jointV) / 2 : 0;

    for (let pair = -1; pair <= cols; pair++) {
      // Stretcher
      ctx.drawImage(materialImg, x, r*(cellH+jointH), stretcherW, cellH);
      x += stretcherW + jointV;

      // Header
      ctx.drawImage(materialImg, x, r*(cellH+jointH), headerW, cellH);
      x += headerW + jointV;

      if (x > canvasW + stretcherW) break;
    }
  }
}
Step 7 — Seeded RNG for tone variation (critical for seamless)
utils/rng.js▼
Every tile must get the same brightness offset every render pass. If you use Math.random(), the tile at position (0,1) gets a different brightness value each render — the seam flickers. Use a deterministic seed based on row+col.
// utils/rng.js

// Mulberry32 — fast, good distribution, deterministic
export function seededRandom(seed) {
  let t = seed + 0x6D2B79F5;
  t = Math.imul(t ^ t >>> 15, t | 1);
  t ^= t + Math.imul(t ^ t >>> 7, t | 61);
  return ((t ^ t >>> 14) >>> 0) / 4294967296;
}

// Usage in render loop:
// const seed = row * 9973 + col * 9871;   // unique per tile position
// const brightnessDelta = (seededRandom(seed) - 0.5) * toneVariation;

// Why these multipliers (9973, 9871)?
// They are primes. Using primes avoids visual patterns in the grid
// (e.g. diagonal stripes) that appear when you use non-prime multipliers
// like row*100 + col, where tiles with row=1,col=0 and row=0,col=10
// produce the same seed (100).
